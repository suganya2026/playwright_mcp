

# Install Node.js via Homebrew
brew install node

# Install Playwright and MCP server
npm install -g playwright @playwright/mcp

# Install Playwright browsers
npx playwright install

# Update Cursor mcp.json
mkdir -p ~/.cursor
nano  ~/.cursor/mcp.json

{
  "mcpServers": {
    "playwright": {
      "command": "npx",
      "args": [
        "-y",
        "@playwright/mcp"
      ]
    }
  }
}
